// 在 startScanning 函数中替换或增加以下逻辑： async function startScanning() { document.getElementById('scanUI').style.display = 'flex'; try { const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent); const devices = await zxing.listVideoInputDevices(); let backId = devices.find(d => d.label.toLowerCase().includes('back') || d.label.toLowerCase().includes('后置'))?.deviceId; const constraints = { video: { deviceId: backId, facingMode: "environment", width: { ideal: 1280 }, height: { ideal: 720 }, // iPhone 强化：请求高帧率 frameRate: { ideal: isIOS ? 60 : 30 } } }; zxing.decodeFromConstraints(constraints, 'v', (res) => { if (res) { const code = res.getText(); const now = Date.now(); if (code === lastScannedCode && (now - lastScannedTime < 2500)) return; lastScannedCode = code; lastScannedTime = now; handleBarcode(code, true); } }); const videoEl = document.getElementById('v'); setTimeout(() => { const track = videoEl.srcObject?.getVideoTracks()[0]; if (track) { const capabilities = track.getCapabilities(); if (!isIOS) { // 安卓专属：2.5秒强力对焦 setInterval(() => { if (capabilities.focusMode?.includes('continuous')) { track.applyConstraints({advanced: [{focusMode: 'continuous'}]}).catch(()=>{}); } }, 2500); } else { // iOS 专属：锁定曝光提升对比度，iPhone 扫码更稳 track.applyConstraints({ advanced: [{ exposureMode: 'continuous', whiteBalanceMode: 'continuous' }] }).catch(()=>{}); } } }, 1000); } catch (err) { showMsg("相机启动失败"); stopScanning(); } }
